129void yycount()
130{
131 for (int i = 0; yytext[i] != '\0'; ++i)
132 {
133 yyline += yytext[i];
134 if (yytext[i] == '\n')
135 {
136 yycolumn = 0;
137 yyline.clear();
138 }
139 else if (yytext[i] == '\t')
140 yycolumn += 8 - (yycolumn % 8);
141 else
142 ++yycolumn;
143 }
144}
