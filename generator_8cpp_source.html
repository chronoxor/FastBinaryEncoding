38#if defined(_WIN32) || defined(_WIN64) || defined(__CYGWIN__)
39 // Sometimes it happens that Visual Studio opens generated files on a fly to analyze.
40 // It opens them with mapping into its process with CreateFileMapping/MapViewOfFile.
41 // As the result files cannot be replaced or removed. Therefore we need to have some
42 // retry attempts with small delay to give a change for the file to be created.
43 // https://stackoverflow.com/questions/41844842/when-error-1224-error-user-mapped-file-occurs
44 const int attempts = 1000;
45 const int sleep = 100;
46 for (int attempt = 0; attempt < attempts; ++attempt)
47 {
48 try
49 {
50 // Rename the unique file inside a loop with retries
51 CppCommon::Path::Rename(unique, filename);
52 return;
53 }
54 catch (const CppCommon::FileSystemException& ex)
55 {
56 if ((ex.system_error() == ERROR_ACCESS_DENIED) || (ex.system_error() == ERROR_USER_MAPPED_FILE))
57 {
58 CppCommon::Thread::Sleep(sleep);
59 continue;
60 }
61 throw;
62 }
63 }
64 throwex CppCommon::FileSystemException("Cannot generate the output file!").Attach(filename);
